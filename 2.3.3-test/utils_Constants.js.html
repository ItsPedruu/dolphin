/**
 * Options for a Dolphin Client
 * @typedef {ClientOptions}
 * @property {?boolean} allowBots Whether to allow bots to execute commands. Default: false
 * @property {?string} prefix The default bot prefix for commands. Default: !
 * @property {?boolean} mentionAsPrefix Whether to use the bot's mention as a prefix. Default: true
 * @property {?array} owners An array of the bot's owners. This is normally used to set commands as owner only. Default: Empty Array
 * @property {?CommandsRegisterOptions} commands Options to register the commands
 * @property {?string} eventsPath The string path to the events folder
 * @property {?boolean} disableReadyMessage Whether to display a console.log message saying the bot is online
 * @property {?boolean} useSendEmbed Whether the text should appear in a nice embed when executing #Message.say() built-in function
 * @property {?string} mainColor The main color used in the embeds for #Message.say() built-in function
 * @property {?string} defaultCommand The default command that displays when the bot is mentioned
 * @property {?string} enginesPath The string path to the engines folder
 */

exports.ClientOptions = {
    "allowBots": false,
    "prefix": "!",
    "mentionAsPrefix": true,
    "owners": [],
    "commands": {
        "path": null,
        "cooldown": 3,
        "defaultCommands": true,
        "useDefaultHandler": true
    },
    "disableReadyMessage": false,
    "useSendEmbed": true,
    "mainColor": "#FFFFFF",
    "defaultCommand": "breefinfo"
}

/**
 * Options to register the commands
 * @typedef {CommandsRegisterOptions}
 * @property {string} path The commands path
 * @property {number} cooldown The amount of seconds to wait before execute another commadn
 * @property {boolean} defaultCommands Whether to use default commands such as ping, reload, breefinfo and eval
 * @property {boolean} useDefaultHandler Whether to use the default command handler
 */
exports.CommandsRegisterOptions = {}

/**
 * 
 * @typedef {CommandOptions}
 * @property {string} name The command name
 * @property {?string} description The command description
 * @property {?string} syntax The command syntax
 * @property {?array} aliases The command aliases
 * @property {?number} defaultArgs The number of arguments needed to be passed to the command in order of him to execute
 * @property {?number} argsCount The number of arguments that you want to return. Don't pass any value to this property if you want to get all the arguments
