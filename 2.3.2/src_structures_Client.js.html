    const Discord = require('discord.js');
const Register = require('../classes/Register.js');
const Utils = require('../../utils/utils.js');

const DEFAULT = require('../../default.json');

/**
 * Represents a Dolphin Clietn
 * @class
 */
class Client extends Discord.Client {
    /**
     * @param {ClientOptions} options Options for the Dolphin Client
     */
    constructor(options) {
        super(options);
        this.options = Utils.deepMerge(DEFAULT.clientOptions, options);

        if (this.options.commands.path) Register.registerCommands(this);
        if (this.options.enginesPath) Register.registerEngines(this);
        Register.registerEvents(this);
    }

    /**
     * Executes a Dolphin Engine
     * @param {string} name The Dolphin Engine name
     * @param  {...array} args The arguments needed to be passed to the #Engine.run() engine function
     */
    executeEngine(name, ...args) {
        let engineData = this.engines.find(engine => engine.name == name);
        if (!engineData) return;

        let EngineClass = require(engineData.filePath);
        new EngineClass().execute(this, ...args);
    }

    startEngines() {
        if (!this.engines) return;
        let runnableEngines = this.engines.filter(engine => engine.runAtStart !== false);

        runnableEngines.forEach(engine => {
            let EngineClass = require(engine.filePath);
            new EngineClass().execute(this);
        });
    }

    /**
     * Add extra features to the discord.js message object
     * @param {Message} message The discord.js message object
     */
    addFeaturesToMessage(message) {
        message.dmVersion = !message.guild;

        message.userFooter = [message.author.username, message.author.avatarURL];

        message.say = async text => {
            let data = this.options.useSendEmbed ? 
                new Discord.RichEmbed().setDescription(text).setColor(this.options.mainColor).setTimestamp().setFooter(...message.userFooter) :
                text;

            return await message.channel.send(data);
        }

        return message;
    }
}

module.exports = Client;
